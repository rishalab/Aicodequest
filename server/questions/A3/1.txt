// Remove Duplicates from a Sorted Array (with A3 bug)
#include <stdio.h>

int remove_duplicates(int arr[], int n) {
    int j = 0;
    for(int i = 1; i < n; i++) {
        if(arr[i] > arr[j]) {
            j++;
            arr[j] = arr[i];
        }
    }
    return j + 1;
}

int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    printf("%d", remove_duplicates(arr, n));
    return 0;
}
