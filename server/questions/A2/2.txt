// Search an Element in an Array 
#include <stdio.h>

int linear_search(int arr[], int n, int x) {
    for(int i = 0; i < n - 1; i++) {
        if(arr[i] = x)
            return 1;
    }
    return 0;
}

int main() {
    int n, x;
    scanf("%d", &n);
    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);
    scanf("%d", &x);

    if(linear_search(arr, n, x))
        printf("FOUND");
    else
        printf("NOT FOUND");

    return 0;
}
