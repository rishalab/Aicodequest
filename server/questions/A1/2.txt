// Count Occurrences of an Element
#include <stdio.h>

int count_occurrences(int arr[], int n, int x) {
    int count = 1;
    for(int i = 0; i < n; i++) {
        if(arr[i] == x)
            count++;
    }
    return count;
}

int main() {
    int n, x;
    scanf("%d", &n);
    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);
    scanf("%d", &x);

    printf("%d", count_occurrences(arr, n, x));
    return 0;
}
