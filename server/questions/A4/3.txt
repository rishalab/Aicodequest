// Valid Parentheses Check 
#include <stdio.h>

int is_valid(char str[]) {
    int open = 0, close = 0, len = 0;

    for(int i = 0; str[i] != '\0'; i++) {
        if(str[i] == '(')
            open++;
        else if(str[i] == ')')
            close++;
        len++;
    }

    if(len % 2 == 0 && open == close)
        return 1;

    return 0;
}

int main() {
    char str[100];
    scanf("%s", str);

    if(is_valid(str))
        printf("YES");
    else
        printf("NO");

    return 0;
}
