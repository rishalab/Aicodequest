// Implement Stack Using Array (A4: Hallucinated Semantics)
#include <stdio.h>

#define MAX 100

int stack[MAX];
int top = -1;

void push(int x) {
    if (top < MAX - 1) {
        stack[++top] = x;
    }
}

void pop() {
    if (top >= 0) {
        top--;
    }
}

int main() {
    int n, k;

    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        int x;
        scanf("%d", &x);
        push(x);
    }

    scanf("%d", &k);

    for (int i = 0; i < k; i++) {
        pop();
    }

    printf("%d", n - k);

    return 0;
}
