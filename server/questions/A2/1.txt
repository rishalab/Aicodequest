// Check if an Array is Sorted 
#include <stdio.h>

int is_sorted(int arr[], int n) {
    for(int i = 1; i < n - 1; i++) {
        if(arr[i] > arr[i-1])
            return 0;
    }
    return 1;
}

int main() {
    int n;
    scanf("%d", &n);
    int arr[n];
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    if(is_sorted(arr, n))
        printf("YES");
    else
        printf("NO");

    return 0;
}
